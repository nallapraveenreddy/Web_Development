### Git Commands
-- git init
-- git add . (Adds Entire Folder)/ git add "Filename"
-- git reset "File name" (revert the file to prevent from commit)
-- git commit -m "Commit message"
-- git push origin main

-- git branch (To list the branches)
-- git branch "new branch name"
-- gi checkout -b "new branch name" (To create and switch to new branch)
-- git checkout main (To switch to main/existing branch)
-- git merge "Branch name" (switch to main branch and merge branch to main)

-- git status (To know the changes made)
-- git log (History of commit)

-- create .gitignore file (To ignore specific files to be pushed into github)
